data(ny_noaa)
df_ny_noaa = 
  ny_noaa %>%
  janitor::clean_names() %>%
  separate(date, into = c("year", "month", "day"), sep = "-") %>%
  mutate(tmax = as.numeric(tmax),
         tmin = as.numeric(tmin),
         prcp = prcp/10
         ) %>%
  mutate(tmax=tmax/10,
         tmin = tmin/10)

Chart A

Boxplot = df_ny_noaa %>%
  filter((month == "01" | month == "07") & !is.na(tmax)) %>%
  group_by(month, year, id) %>%
  summarize(mean_temp = mean(tmax)) %>% 
  ggplot(aes(x = year, y = mean_temp)) +
  geom_boxplot() +
  facet_grid(.~month) +
  theme(axis.text.x = element_text(angle = 90, hjust = 1)) +
  labs(
    title = "Average maximum temperature in January and in July in each station across years",
    x = "Year",
    y = "Average temperature"
  )
ggplotly(Boxplot)

df_ny_noaa %>%
  group_by(year) %>%
  summarise(snow_sum = sum(snow, na.rm = T)) %>%
  plot_ly(x = ~year, y = ~snow_sum, type = "scatter", mode = "makers",
          alpha = 0.5) %>%
  layout(
    title = "Total snow for years",
    scene = list(
      xaxis = list(title = "Year"),
      yaxis = list(title = "Snow")
